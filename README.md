# Grammar(byacc style)
LEFT { RIGHT }  EQ =

program:
    | program _list

_list: _var1 
    | _var2 EQ _val1
    | _var2 EQ _left _val2 _right 
    | _var2 EQ LEFT RIGHT 
    | LEFT RIGHT
    
_val1: STR
    
_val2: _val2 _var1  
    | _var1 
    | _var2 EQ _left _val2 _right 
    | _val2 _var2 EQ _left _val2 _right  
    | _var2 EQ LEFT RIGHT 
    | _val2 _var2 EQ LEFT RIGHT  
    | _var2 EQ _val1 
    | _val2 _var2 EQ _val1  
    | LEFT RIGHT 
    | _val2 LEFT RIGHT  
    | LEFT _val2 RIGHT
    | _val2 LEFT _val2 RIGHT

_var1: STR

_var2: STR 

_left: LEFT 

_right: RIGHT
